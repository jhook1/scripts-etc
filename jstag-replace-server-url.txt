<script type="text/javascript">
	// Retrieve server relative URL from config for client.
	// Can execute via [string literal].replace(/(\.\.\/)+/g, typeof getServerUrl === 'undefined' ? '$&' : getServerUrl())
	// Replaces file relative paths (../) with server relative path (/Apps/Portal/..) if this function is defined.
	function getServerUrl() {
		return '<%=ResolveUrl(ConfigurationManager.AppSettings("PVS"))%>';
	}
</script>
